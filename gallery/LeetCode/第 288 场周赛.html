class Solution {
public:

    int largestInteger(int num) {
        int n = 0, m = 0;
        int st[10];

            vector<int> old, even;
            n = num;
            int tmp = n, cnt = 0;
            while (tmp) {
                int c = tmp % 10;
                if (c & 1) {
                    old.push_back(c);
                }
                else
                    even.push_back(c);
                tmp /= 10;
                st[cnt++] = c & 1;
            }

            sort(old.begin(), old.end(), [=](int a, int b) { return a > b; });
            sort(even.begin(), even.end(), [=](int a, int b) { return a > b; });

            int a = 0, b = 0;
            for (int i = cnt - 1; i >= 0; i--) {
                m *= 10;
                if (st[i]) {
                     m += old[a++];
                }
                else
                    m += even[b++];
            }
            return m;
    }
};

class Solution {
public:
    string minimizeResult(string expression) {
        string num1, num2;
        for (int i = 0; i < expression.size(); i ++ ) {
            if (expression[i] == '+') {
                num1 = expression.substr(0, i);
                num2 = expression.substr(i + 1);
                break;
            }
        }

        int minv = atoi(num1.c_str()) + atoi(num2.c_str());
        string minstr = "(" + expression + ")";

        for (int i = 0; i < num1.size(); i ++ ) {
            for (int j = 0; j < num2.size(); j ++ ) {
                int l1, l2, r1, r2, sum;
                string sl1, sr2;
                if (i == 0) {
                    l1 = 1;
                    sl1 = "";
                }
                else {
                    l1 = atoi(num1.substr(0, i).c_str());
                    sl1 = num1.substr(0, i);
                }
                if (j == num2.size() - 1) {
                    r2 = 1;
                    sr2 = "";
                }
                else {
                    r2 = atoi(num2.substr(j + 1, num2.size() - j - 1).c_str());
                    sr2 = num2.substr(j + 1, num2.size() - j - 1);
                }

                l2 = atoi(num1.substr(i, num1.size() - i).c_str());
                r1 = atoi(num2.substr(0, j + 1).c_str());

                sum = l1 * (l2 + r1) * r2;
                if (sum < minv) {
                    minv = sum;
                    minstr = sl1 + "(" + num1.substr(i, num1.size() - i) + "+"
                        + num2.substr(0, j + 1) + ")" + sr2;
                }
            }
        }
        return minstr;
    }
};

小根堆的题目

class Solution {
public:
    static const int MOD = 1e9 + 7;

    int maximumProduct(vector<int>& nums, int k) {
        priority_queue<int,vector<int>,greater<int> > heap;
        for (auto &c : nums) {
            heap.push(c);
        }
        while (k -- ) {
            int t = heap.top(); heap.pop();
            t ++ ;
            heap.push(t);
        }

        long long res = 1;
        while (heap.size()) {
            res = (res * heap.top()) % MOD; heap.pop();
        }
        return res;
    }
};
